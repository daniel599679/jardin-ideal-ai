function selectSize(btn){
  document.querySelectorAll('.est-size-card').forEach(b => b.classList.remove('active'));

  updateServicePrices();

// Mettre à jour les prix affichés sur chaque carte de service
function updateServicePrices(){
  const sz = estCurrentSize;
  Object.keys(PRICE_GRID).forEach(svc => {
    const el = document.getElementById('price-' + svc);
    if(!el) return;
    const basePrice = PRICE_GRID[svc][sz];
    el.textContent = 'à partir de ' + basePrice.toLocaleString('fr-CA') + ' $';
  });
}

function calcEst(){
  const sz = estCurrentSize || 'M';
  const mult = parseFloat(document.getElementById('estType').value) || 1;
  const checks = document.querySelectorAll('#estChecks input:checked');

  if(checks.length === 0){
    document.getElementById('estVal').textContent = 'Sélectionnez des éléments';
    document.getElementById('estNote').textContent = 'Estimation indicative — soumission gratuite sans engagement';
    return;
  }

  let base = 0;
  checks.forEach(c => {
    const svc = c.dataset.service;
    if(svc && PRICE_GRID[svc]) base += PRICE_GRID[svc][sz];
  });

  // Appliquer le multiplicateur de type de propriété
  const low  = Math.round(base * mult * 0.90 / 500) * 500;
  const high = Math.round(base * mult * 1.35 / 500) * 500;

  document.getElementById('estVal').textContent =
    low.toLocaleString('fr-CA') + ' $ — ' + high.toLocaleString('fr-CA') + ' $';

  const items = [...checks].map(c => c.dataset.label).join(', ');
  const sizeLabels = { S:'petite cour', M:'cour moyenne', L:'grande cour', XL:'très grande cour' };
  document.getElementById('estNote').textContent =
    'Inclus : ' + items + ' · ' + sizeLabels[sz] + ' · Estimation indicative — prix réel établi après visite gratuite';

  // Afficher le mini-calculateur
  const midAmt = Math.round((low + high) / 2 / 500) * 500;
  window._miniCalcAmt = midAmt;
  window._miniCalcLow = low;
  window._miniCalcHigh = high;
  const amtStr = midAmt.toLocaleString('fr-CA') + ' $';
  document.getElementById('miniCalcAmt').textContent = amtStr;
  const inlineEl = document.getElementById('miniCalcAmtInline');
  if(inlineEl) inlineEl.textContent = amtStr;
  document.getElementById('miniCalc').style.display = 'block';
  updateCapitalSlider();
  miniCalcCompute();
  updateExamples();
}

function buildProgress(){

function selectOpt(card, qId, val, type, evt){
  if(evt){ evt.preventDefault(); evt.stopPropagation(); }
  // Feedback visuel instantané
  card.style.transform = 'scale(.96)';
  requestAnimationFrame(()=>{ card.style.transform = ''; });
  if(type==='radio'){
    document.querySelectorAll('#stepContent .opt-card').forEach(c=>c.classList.remove('selected'));
    card.classList.add('selected');
    answers[qId] = val;
    setTimeout(()=>goNext(), 220);
  } else {
    card.classList.toggle('selected');
    if(!answers[qId]) answers[qId]=[];
    const idx = answers[qId].indexOf(val);
    if(idx>-1) answers[qId].splice(idx,1);
    else answers[qId].push(val);
  }
}

function goNext(){
  const q = QUESTIONS[currentStep];
  const hasAnswer = q.type==='multi'
    ? (answers[q.id]&&answers[q.id].length>0)
    : !!answers[q.id];
  if(!hasAnswer){
    const btn = document.getElementById('btnNext');
    btn.style.animation='none'; btn.offsetHeight;
    btn.style.animation='shake .4s ease';
    return;
  }
  if(currentStep < QUESTIONS.length-1){
    currentStep++;
    renderStep();
  } else {
    showResultMagazine();
  }
}

function goBack(){
  if(currentStep>0){ currentStep--; renderStep(); }
}

  let score=0;
  QUESTIONS.forEach(q=>{
    if(q.type==='multi' && answers[q.id]){
      answers[q.id].forEach(v=>{ const o=q.opts.find(x=>x.val===v); if(o) score+=o.pts; });
    } else if(answers[q.id]){
      const o=q.opts.find(x=>x.val===answers[q.id]); if(o) score+=o.pts;
    }
  });
  const max=100;
  return Math.min(Math.round(score/max*100), 100);
}

function handleFiles(files){
  [...files].forEach(f=>{
    if(f.size>52428800){alert('Fichier trop volumineux (max 50 Mo) : '+f.name);return;}
    uploadedFiles.push(f);
    const reader=new FileReader();
    reader.onload=ev=>{
      const prev=document.getElementById('uploadPreview');
      const div=document.createElement('div'); div.className='up-thumb';
      if(f.type.startsWith('image/')){ const img=document.createElement('img'); img.src=ev.target.result; div.appendChild(img); }
      else if(f.type.startsWith('video/')){ const vid=document.createElement('video'); vid.src=ev.target.result; vid.muted=true; div.appendChild(vid); }
      else { div.style.background='rgba(255,255,255,.1)'; div.style.display='flex'; div.style.alignItems='center'; div.style.justifyContent='center'; div.style.fontSize='.6rem'; div.style.color='rgba(255,255,255,.6)'; div.textContent=f.name.split('.').pop().toUpperCase(); }
      const del=document.createElement('button'); del.className='up-del'; del.textContent='×';
      del.onclick=()=>{ uploadedFiles.splice(uploadedFiles.indexOf(f),1); div.remove(); };
      div.appendChild(del); prev.appendChild(div);
    };
    reader.readAsDataURL(f);
  });
}

// Bouton de navigation dans le chat
function addNavBtn(label, href){
  const msgs = document.getElementById('chatMessages');
  const btn = document.createElement('button');
  btn.className = 'chat-nav-btn';
  btn.innerHTML = `➡️ ${label}`;
  btn.addEventListener('click', () => {
    if(href.startsWith('tel:') || href.startsWith('http')){
      window.open(href, '_blank');
    } else {
      const el = document.querySelector(href);
      if(el) el.scrollIntoView({behavior:'smooth', block:'start'});
      toggleChat();
    }
  });
  msgs.appendChild(btn);
  msgs.scrollTop = msgs.scrollHeight;
}

  const exDiv = document.getElementById('miniExamples');
  const grid = document.getElementById('miniExamplesGrid');
  const tbody = document.getElementById('miniExamplesTableBody');
  if(!exDiv || !grid || !tbody) return;
  if(checks.length === 0){ exDiv.style.display='none'; return; }

  // Map des labels vers clés
  const labelMap = {
    'Patio pavé uni':'pave','Terrasse composite':'terrasse','Pergola / Gazebo':'pergola',
    'Piscine creusée':'piscine','Cuisine extérieure':'cuisine','Clôture & murets':'cloture',
    'Éclairage extérieur':'eclairage','Aménagement paysager':'amenagement'
  };

  const keys = [...checks].map(c=>labelMap[c.dataset.label]).filter(Boolean);
  if(keys.length === 0){ exDiv.style.display='none'; return; }

  // Construire les cartes images
  grid.innerHTML = '';
  keys.forEach(k=>{
    const ex = SERVICE_EXAMPLES[k];
    if(!ex) return;
    [ex.img, ex.img2].forEach(imgSrc=>{
      const card = document.createElement('div');
      card.className = 'mini-ex-card';
      card.innerHTML = `<img src="${imgSrc}" alt="${ex.label}" loading="lazy">
        <div class="mini-ex-label">${ex.label}</div>
        <div class="mini-ex-price">${fmtEx(ex.cost)}</div>`;
      grid.appendChild(card);
    });
  });

  // Construire le tableau
  tbody.innerHTML = '';
  keys.forEach(k=>{
    const ex = SERVICE_EXAMPLES[k];
    if(!ex) return;
    const tr = document.createElement('tr');
    tr.innerHTML = `<td>${ex.label}</td><td>${fmtEx(ex.cost)}</td><td>${fmtEx(ex.row60)}/mois</td><td>${fmtEx(ex.row120)}/mois</td>`;
    tbody.appendChild(tr);
  });

  exDiv.style.display = 'block';
}

function updateCapitalSlider(){
  const low = window._miniCalcLow || 0;
  const high = window._miniCalcHigh || 0;
  if(!low && !high) return;
  const slider = document.getElementById('capitalSlider');
  if(!slider) return;
  slider.min = 0;
  slider.max = high;
  slider.step = Math.max(500, Math.round(high / 100 / 500) * 500);
  slider.value = 0;
  document.getElementById('capitalDisplay').textContent = '0 $';
  document.getElementById('capitalHintLow').textContent = '0 $';
  document.getElementById('capitalHintHigh').textContent = high.toLocaleString('fr-CA') + ' $';
  document.getElementById('capitalSoldeRow').style.display = 'none';
  // reset gradient
  slider.style.setProperty('--pct', '0%');
}

function onCapitalSlide(){
  const slider = document.getElementById('capitalSlider');
  const capital = parseInt(slider.value) || 0;
  const high = window._miniCalcHigh || 0;
  // Update gradient
  const pct = high > 0 ? Math.round(capital / high * 100) : 0;
  slider.style.setProperty('--pct', pct + '%');
  document.getElementById('capitalDisplay').textContent = capital.toLocaleString('fr-CA') + ' $';
  // Solde row
  const budget = window._miniCalcAmt || 0;
  const net = Math.max(0, budget - capital);
  if(capital > 0){
    document.getElementById('capitalSoldeRow').style.display = 'block';
    document.getElementById('capitalSoldeCapital').textContent = capital.toLocaleString('fr-CA') + ' $';
    document.getElementById('capitalSoldeNet').textContent = net.toLocaleString('fr-CA') + ' $';
  } else {
    document.getElementById('capitalSoldeRow').style.display = 'none';
  }
  // Recalculate
  miniCalcCompute();
}

function miniCalcCompute(){
  const budget = window._miniCalcAmt || 0;
  if(budget <= 0) return;
  const capitalSlider = document.getElementById('capitalSlider');
  const capital = capitalSlider ? (parseInt(capitalSlider.value) || 0) : 0;
  const net = Math.max(0, budget - capital);
  // Mettre à jour le montant affiché dans la colonne info
  const inlineEl = document.getElementById('miniCalcAmtInline');
  if(inlineEl) inlineEl.textContent = net.toLocaleString('fr-CA') + ' $';
  document.getElementById('miniCalcAmt').textContent = net.toLocaleString('fr-CA') + ' $';
  if(net <= 0){
    document.getElementById('miniMonthly').textContent = '0';
    document.getElementById('miniTotal').textContent = '0 $';
    document.getElementById('miniInterest').textContent = '0 $';
    return;
  }
  const months = parseInt(document.getElementById('miniMonths').value) || 60;
  const RATE = 12.99; // taux fixe entreprise
  const r = RATE / 100 / 12;
  const monthly = net * (r * Math.pow(1+r, months)) / (Math.pow(1+r, months) - 1);
  const totalPay = monthly * months;
  const totalInterest = totalPay - net;
  document.getElementById('miniMonthly').textContent = miniCalcFmt(monthly);
  document.getElementById('miniTotal').textContent = miniCalcFmt(totalPay);
  document.getElementById('miniInterest').textContent = miniCalcFmt(totalInterest);
}

function copyQrLink(){
  const url = 'https://jardin-ideal.com/lp-cour-arriere';
  if(navigator.clipboard && navigator.clipboard.writeText){
    navigator.clipboard.writeText(url).then(function(){
      const btn = document.getElementById('btnCopyQr');
      const orig = btn.innerHTML;
      btn.innerHTML = '<svg viewBox="0 0 20 20" fill="currentColor" width="16" height="16"><path d="M16.7 5.3l-9 9-4-4 1.4-1.4 2.6 2.6 7.6-7.6 1.4 1.4z"/></svg> Lien copié!';
      btn.style.background = 'rgba(0,142,63,.3)';
      btn.style.borderColor = 'var(--green)';
      setTimeout(function(){ btn.innerHTML = orig; btn.style.background = ''; btn.style.borderColor = ''; }, 2500);
    });
  } else {
    // Fallback pour anciens navigateurs
    const ta = document.createElement('textarea');
    ta.value = url;
    document.body.appendChild(ta);
    ta.select();
    document.execCommand('copy');
    document.body.removeChild(ta);
    alert('Lien copié : ' + url);
  }
}

function openPrequalModal() {
  // Pré-remplir montant depuis l'estimateur
  const montantEl = document.getElementById('pqMontant');
  if (montantEl && window._miniCalcAmt) {
    montantEl.value = window._miniCalcAmt;
  }
  // Pré-remplir service depuis les cases cochées
  const checked = Array.from(document.querySelectorAll('.est-cb input:checked')).map(cb => {
    const lbl = cb.closest('label');
    return lbl ? lbl.textContent.trim().replace(/^[^\w\u00C0-\u017E]+/, '') : '';
  }).filter(Boolean);
  if (checked.length > 0) {
    const svc = document.getElementById('pqService');
    if (svc) {
      const match = Array.from(svc.options).find(o =>
        checked.some(c => o.text.toLowerCase().includes(c.toLowerCase().split(' ')[0]))
      );
      if (match) svc.value = match.value;
    }
  }
  // ── AUTO-REMPLISSAGE ──
  // 1. Depuis les paramètres URL (Facebook Lead Ads, Google Ads, liens directs)
  const urlParams = new URLSearchParams(window.location.search);
  const pqMap = {
    pqPrenom: ['first_name','prenom','fname','given_name'],
    pqNom:    ['last_name','nom','lname','family_name','surname'],
    pqTel:    ['phone','telephone','tel','mobile','phone_number'],
    pqEmail:  ['email','courriel','email_address'],
    pqVille:  ['city','ville','address_city','location']
  };
  Object.entries(pqMap).forEach(([fieldId, keys]) => {
    const el = document.getElementById(fieldId);
    if (!el || el.value) return; // Ne pas écraser une valeur déjà saisie
    for (const key of keys) {
      const val = urlParams.get(key) || urlParams.get(key.replace(/_/g,'-'));
      if (val) { el.value = decodeURIComponent(val); break; }
    }
  });

  // 2. Depuis le formulaire principal de la page (si déjà rempli)
  const mainFields = {
    pqPrenom: ['fPrenom','prenom','first_name'],
    pqNom:    ['fNom','nom','last_name'],
    pqTel:    ['fTel','telephone','phone'],
    pqEmail:  ['fEmail','courriel','email'],
    pqVille:  ['fVille','ville','city']
  };
  Object.entries(mainFields).forEach(([fieldId, keys]) => {
    const el = document.getElementById(fieldId);
    if (!el || el.value) return;
    for (const key of keys) {
      const mainEl = document.getElementById(key) || document.querySelector('[name="'+key+'"]');
      if (mainEl && mainEl.value.trim()) { el.value = mainEl.value.trim(); break; }
    }
  });

  // 3. Depuis localStorage (session précédente ou lead Facebook pixel)
  try {
    const saved = JSON.parse(localStorage.getItem('ji_user_info') || '{}');
    const lsMap = { pqPrenom:'prenom', pqNom:'nom', pqTel:'telephone', pqEmail:'courriel', pqVille:'ville' };
    Object.entries(lsMap).forEach(([fieldId, key]) => {
      const el = document.getElementById(fieldId);
      if (el && !el.value && saved[key]) el.value = saved[key];
    });
  } catch(e) {}

  // Afficher étape 1
  ['pqStep1','pqStep2','pqStep3','pqConfirm'].forEach(id => {
    const el = document.getElementById(id);
    if (el) el.style.display = (id === 'pqStep1') ? '' : 'none';
  });
  const bar = document.getElementById('pqStepBar');
  if (bar) bar.style.display = 'flex';
  pqSetDot(1);
  document.getElementById('prequalModal').style.display = 'block';
  document.body.style.overflow = 'hidden';
}

function closePrequalModal() {
  document.getElementById('prequalModal').style.display = 'none';
  document.body.style.overflow = '';
}

function pqSetDot(step) {
  for (let i = 1; i <= 3; i++) {
    const dot = document.getElementById('pqDot' + i);
    if (!dot) continue;
    dot.classList.remove('active','done');
    if (i < step) dot.classList.add('done');
    else if (i === step) dot.classList.add('active');
  }
}

function pqNext(step) {
  if (step === 1) {
    const prenom = document.getElementById('pqPrenom').value.trim();
    const nom = document.getElementById('pqNom').value.trim();
    const tel = document.getElementById('pqTel').value.trim();
    const email = document.getElementById('pqEmail').value.trim();
    const ville = document.getElementById('pqVille').value.trim();
    if (!prenom || !nom || !tel || !email || !ville) {
      alert('Veuillez remplir tous les champs obligatoires.');
      return;
    }
    document.getElementById('pqStep1').style.display = 'none';
    document.getElementById('pqStep2').style.display = '';
    pqSetDot(2);
  } else if (step === 2) {
    const service = document.getElementById('pqService').value;
    const montant = document.getElementById('pqMontant').value;
    if (!service || !montant) {
      alert('Veuillez indiquer le service souhaité et le montant estimé.');
      return;
    }
    document.getElementById('pqStep2').style.display = 'none';
    document.getElementById('pqStep3').style.display = '';
    pqSetDot(3);
  }
  const modal = document.getElementById('prequalModal');
  if (modal) modal.scrollTop = 0;
}

function pqBack(step) {
  if (step === 2) {
    document.getElementById('pqStep2').style.display = 'none';
    document.getElementById('pqStep1').style.display = '';
    pqSetDot(1);
  } else if (step === 3) {
    document.getElementById('pqStep3').style.display = 'none';
    document.getElementById('pqStep2').style.display = '';
    pqSetDot(2);
  }
  const modal = document.getElementById('prequalModal');
  if (modal) modal.scrollTop = 0;
}

<script>
// ============================================================
// ============================================================
// ALEX VOICE ENGINE v2 — Moteur vocal haute qualité
// Architecture: segmentation par phrases + file d'attente
// Qualité: magazine radio, rythme naturel, pauses intelligentes
// ============================================================

// =============================================================
// ALEX VOICE ENGINE v5 — MOTEUR HUMAIN HAUTE FIDÉLITÉ
// Voix naturelle, chaleur humaine, rythme conseiller professionnel
// =============================================================

// ---- État global ----
let alexVoiceEnabled = true;
let alexSpeaking = false;
let alexSpeechRecog = null;
let alexListening = false;
let alexVoiceQueue = [];
let alexVoicePlaying = false;
let alexSelectedVoice = null;
let alexVoicesLoaded = false;
let alexChromeFixInterval = null;

// ---- Profils vocaux dynamiques ----
// Chaque type de phrase a ses propres paramètres pour imiter les inflexions humaines
const ALEX_VOICE_PROFILES = {
  // Phrase normale — ton chaleureux de conseiller
  normal:    { rate: 0.92, pitch: 1.05, volume: 1.0 },
  // Question — légère montée de ton, curiosité naturelle
  question:  { rate: 0.90, pitch: 1.12, volume: 1.0 },
  // Emphase — légèrement plus lent, plus grave, plus d'autorité
  emphasis:  { rate: 0.82, pitch: 0.98, volume: 1.0 },
  // Chiffre / prix — clair, distinct, bien articulé
  number:    { rate: 0.80, pitch: 1.00, volume: 1.0 },
  // Introduction / accueil — chaleureux, souriant
  greeting:  { rate: 0.94, pitch: 1.10, volume: 1.0 },
  // Conclusion / CTA — confiant, invitant
  cta:       { rate: 0.88, pitch: 1.02, volume: 1.0 },
  // Confidence builder — posé, rassurant
  trust:     { rate: 0.85, pitch: 0.99, volume: 1.0 },
  // Citation — légèrement plus lent, ton narratif
  quote:     { rate: 0.80, pitch: 0.97, volume: 0.98 }
};

// ---- Pauses naturelles (ms) ----
const ALEX_PAUSES = {
  breath:    120,   // Micro-pause respiratoire entre mots
  comma:     220,   // Virgule
  semicolon: 300,   // Point-virgule
  sentence:  420,   // Fin de phrase
  paragraph: 600,   // Changement de sujet
  dramatic:  750,   // Pause dramatique avant info importante
  question:  480    // Après une question
};

// ---- Priorité des voix (ordre de préférence) ----
const ALEX_VOICE_PRIORITY = [
  // macOS/iOS — les plus naturelles
  v => v.name === 'Thomas' && v.lang === 'fr-FR',
  v => v.name === 'Amelie' && v.lang === 'fr-CA',
  v => v.name === 'Amélie' && v.lang === 'fr-CA',
  v => v.name.includes('Thomas') && v.lang.startsWith('fr'),
  v => v.name.includes('Amélie') || v.name.includes('Amelie'),
  v => v.name.includes('Virginie'),
  v => v.name.includes('Audrey'),
  // Windows — Microsoft Neural (Edge/Chrome)
  v => v.name.includes('Microsoft') && v.name.includes('Henri') && v.lang.startsWith('fr'),
  v => v.name.includes('Microsoft') && v.name.includes('Denise') && v.lang.startsWith('fr'),
  v => v.name.includes('Microsoft') && v.name.includes('Sylvie') && v.lang.startsWith('fr'),
  v => v.name.includes('Hortense'),
  v => v.name.includes('Julie') && v.lang.startsWith('fr'),
  v => v.name.includes('Paul') && v.lang.startsWith('fr'),
  // Android / Chrome
  v => v.lang === 'fr-CA' && v.localService,
  v => v.lang === 'fr-FR' && v.localService,
  v => v.lang === 'fr-CA',
  v => v.lang === 'fr-FR',
  v => v.lang.startsWith('fr')
];

// ---- Charger et mettre en cache la meilleure voix ----
function alexLoadVoice(){
  const voices = window.speechSynthesis.getVoices();
  if(!voices || voices.length === 0) return null;
  for(const test of ALEX_VOICE_PRIORITY){
    const found = voices.find(test);
    if(found){ alexSelectedVoice = found; return found; }
  }
  alexSelectedVoice = voices[0] || null;
  return alexSelectedVoice;
}

// Pré-charger les voix dès que possible
if(window.speechSynthesis){
  alexLoadVoice();
  window.speechSynthesis.addEventListener('voiceschanged', function(){
    alexLoadVoice();
    alexVoicesLoaded = true;
  });
  setTimeout(function(){
    if(!alexSelectedVoice) alexLoadVoice();
    alexVoicesLoaded = true;
  }, 800);
}

// ---- Nettoyer le texte pour la synthèse vocale ----
function alexCleanText(text){
  return text
    .replace(/[\u{1F000}-\u{1FFFF}]/gu, '')
    .replace(/[\u2600-\u27BF]/g, '')
    .replace(/[\u{1F300}-\u{1F9FF}]/gu, '')
    .replace(/➡️|🔵|📞|✉️|📍|🌟|😊|👋|🧧|🌳|⛺|🏊|🍽️|🚧|💡|1️⃣|2️⃣|3️⃣|🔴|🟢|⭐|✅|❌/g, '')
    .replace(/\$/g, ' dollars')
    .replace(/%/g, ' pourcent')
    .replace(/RBQ/g, 'R B Q')
    .replace(/2M\$/g, 'deux millions de dollars')
    .replace(/(\d+)h(\d+)?/g, (m, h, min) => min ? `${h} heures ${min}` : `${h} heures`)
    .replace(/\n[-•]\s*/g, ', ')
    .replace(/\n\d+\.\s*/g, ', ')
    .replace(/\n+/g, ' ')
    .replace(/\s{2,}/g, ' ')
    .trim();
}

// ---- Détecter le profil vocal d'un segment ----
// Analyse le contenu pour choisir les bons paramètres pitch/rate
function alexDetectProfile(text){
  const t = text.toLowerCase();
  // Question
  if(text.trim().endsWith('?')) return 'question';
  // Chiffre / prix
  if(/\d{3,}|dollars|pourcent|mois|ans|heures/.test(t)) return 'number';
  // Accueil / salutation
  if(/bonjour|enchanté|bienvenue|super|excellent|magnifique|parfait|génial/.test(t)) return 'greeting';
  // Citation
  if(/disait|dit toujours|citation|comme|selon/.test(t)) return 'quote';
  // Confiance / garantie
  if(/garantie|rbq|licencié|assurance|confiance|certif|sécurité/.test(t)) return 'trust';
  // CTA / invitation à l'action
  if(/allons|commençons|remplissez|cliquez|appelez|contactez|soumission|gratuit/.test(t)) return 'cta';
  // Emphase
  if(/important|attention|rappel|jamais|toujours|meilleur|unique|exclusif/.test(t)) return 'emphasis';
  return 'normal';
}

// ---- Créer un utterance avec profil vocal adapté ----
function alexCreateUtterance(text, profileOverride){
  const profile = profileOverride || alexDetectProfile(text);
  const params = ALEX_VOICE_PROFILES[profile] || ALEX_VOICE_PROFILES.normal;

  const utt = new SpeechSynthesisUtterance(text);
  utt.lang = 'fr-CA';
  utt.rate   = params.rate;
  utt.pitch  = params.pitch;
  utt.volume = params.volume;

  const voice = alexSelectedVoice || alexLoadVoice();
  if(voice) utt.voice = voice;

  return utt;
}

// ---- Jouer la file d'attente segment par segment ----
function alexPlayQueue(){
  if(alexVoicePlaying || alexVoiceQueue.length === 0) return;
  if(!alexVoiceEnabled || !window.speechSynthesis) return;

  alexVoicePlaying = true;
  alexSpeaking = true;

  const avatar = document.getElementById('chatAvatar');
  const wave   = document.getElementById('chatWaveBar');
  if(avatar) avatar.classList.add('speaking');
  if(wave)   wave.classList.add('active');

  const segment = alexVoiceQueue.shift();
  const utt = alexCreateUtterance(segment.text, segment.profile);

  utt.onend = function(){
    const pauseMs = segment.pause !== undefined ? segment.pause : ALEX_PAUSES.sentence;
    setTimeout(function(){
      alexVoicePlaying = false;
      if(alexVoiceQueue.length > 0){
        alexPlayQueue();
      } else {
        alexSpeaking = false;
        if(avatar) avatar.classList.remove('speaking');
        if(wave)   wave.classList.remove('active');
      }
    }, pauseMs);
  };

  utt.onerror = function(e){
    if(e.error === 'interrupted' || e.error === 'canceled') return;
    alexVoicePlaying = false;
    alexVoiceQueue   = [];
    alexSpeaking     = false;
    if(avatar) avatar.classList.remove('speaking');
    if(wave)   wave.classList.remove('active');
  };

  alexChromeFix();
  window.speechSynthesis.speak(utt);
}

// ---- Workaround Chrome: bug speechSynthesis ----
function alexChromeFix(){
  if(alexChromeFixInterval) return;
  alexChromeFixInterval = setInterval(function(){
    if(!window.speechSynthesis.speaking){
      clearInterval(alexChromeFixInterval);
      alexChromeFixInterval = null;
      return;
    }
    window.speechSynthesis.pause();
    window.speechSynthesis.resume();
  }, 10000);
}

// ---- alexSpeak(text) : API de base ----
function alexSpeak(text, options){
  if(!alexVoiceEnabled || !window.speechSynthesis) return;
  const opts = options || {};

  if(opts.interrupt){
    window.speechSynthesis.cancel();
    alexVoiceQueue   = [];
    alexVoicePlaying = false;
    alexSpeaking     = false;
    if(alexChromeFixInterval){ clearInterval(alexChromeFixInterval); alexChromeFixInterval = null; }
    setTimeout(function(){ alexSpeakNow(text, opts); }, 200);
    return;
  }
  alexSpeakNow(text, opts);
}

function alexSpeakNow(text, opts){
  const clean = alexCleanText(text);
  if(!clean) return;
  // Découper en phrases naturelles
  const sentences = clean.split(/(?<=[.!?])\s+/);
  sentences.forEach(s => {
    s = s.trim();
    if(!s) return;
    alexVoiceQueue.push({ text: s, pause: ALEX_PAUSES.sentence });
  });
  if(!alexVoicePlaying) alexPlayQueue();
}

// ---- alexSpeakSequence(segments) : séquence avec contrôle total ----
// Chaque segment peut avoir: text, pause, profile
function alexSpeakSequence(segments){
  if(!alexVoiceEnabled || !window.speechSynthesis) return;

  window.speechSynthesis.cancel();
  alexVoiceQueue   = [];
  alexVoicePlaying = false;

  setTimeout(function(){
    segments.forEach(seg => {
      if(typeof seg === 'string'){
        const clean = alexCleanText(seg);
        if(clean) alexVoiceQueue.push({ text: clean, pause: ALEX_PAUSES.sentence });
      } else {
        const clean = alexCleanText(seg.text || '');
        if(clean) alexVoiceQueue.push({
          text:    clean,
          pause:   seg.pause !== undefined ? seg.pause : ALEX_PAUSES.sentence,
          profile: seg.profile || null
        });
      }
    });
    alexPlayQueue();
  }, 200);
}

// ---- toggleAlexVoice() : activer/désactiver la voix ----
function toggleAlexVoice(){
  alexVoiceEnabled = !alexVoiceEnabled;
  const btn = document.getElementById('chatVoiceToggle');
  const status = document.getElementById('chatStatus');

  if(alexVoiceEnabled){
    if(btn){ btn.classList.add('active'); btn.title = 'Désactiver la voix d\'Alex'; }
    if(status) status.textContent = '🟢 En ligne · Voix activée';
    // Confirmation vocale courte
    setTimeout(function(){
      alexSpeakSequence([
        { text: 'Voix activée.', pause: ALEX_PAUSES.sentence, profile: 'greeting' },
        { text: 'Je suis là, prêt à vous guider.', pause: 0, profile: 'normal' }
      ]);
    }, 200);
  } else {
    window.speechSynthesis && window.speechSynthesis.cancel();
    alexVoiceQueue = [];
    alexVoicePlaying = false;
    alexSpeaking = false;
    if(alexChromeFixInterval){ clearInterval(alexChromeFixInterval); alexChromeFixInterval = null; }
    const avatar = document.getElementById('chatAvatar');
    const wave = document.getElementById('chatWaveBar');
    if(avatar) avatar.classList.remove('speaking');
    if(wave) wave.classList.remove('active');
    if(btn){ btn.classList.remove('active'); btn.title = 'Activer la voix d\'Alex'; }
    if(status) status.textContent = '🟢 En ligne · Voix désactivée';
  }
}

// ---- toggleSpeechInput() : reconnaissance vocale (mic) ----
function toggleSpeechInput(){
  const micBtn = document.getElementById('chatMicBtn');
  const input = document.getElementById('chatInput');

  const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
  if(!SpeechRecognition){
    addBotMsg('La reconnaissance vocale n\'est pas disponible sur ce navigateur. Essayez Chrome ou Edge.');
    return;
  }

  if(alexListening){
    alexListening = false;
    if(alexSpeechRecog) alexSpeechRecog.stop();
    if(micBtn) micBtn.classList.remove('listening');
    return;
  }

  // Mettre en pause la synthèse vocale pendant l'écoute
  if(alexSpeaking){
    window.speechSynthesis && window.speechSynthesis.pause();
  }

  alexSpeechRecog = new SpeechRecognition();
  alexSpeechRecog.lang = 'fr-CA';
  alexSpeechRecog.interimResults = false;
  alexSpeechRecog.maxAlternatives = 1;
  alexSpeechRecog.continuous = false;

  alexListening = true;
  if(micBtn) micBtn.classList.add('listening');
  if(input) input.placeholder = '🎤 Je vous écoute...';

  alexSpeechRecog.onresult = function(event){
    const transcript = event.results[0][0].transcript;
    if(input) input.value = transcript;
    setTimeout(function(){ sendChat(); }, 400);
  };

  alexSpeechRecog.onerror = function(event){
    if(event.error === 'not-allowed'){
      addBotMsg('L\'accès au microphone a été refusé. Veuillez l\'autoriser dans les paramètres de votre navigateur.');
    }
  };

  alexSpeechRecog.onend = function(){
    alexListening = false;
    if(micBtn) micBtn.classList.remove('listening');
    if(input) input.placeholder = 'Écrivez ou parlez à Alex...';
    if(alexSpeaking){
      window.speechSynthesis && window.speechSynthesis.resume();
    }
  };

  alexSpeechRecog.start();
}
